#!/usr/bin/env python3
"""
Daily Security News Video Shot List Generator
==============================================
Production-grade module to define, validate, and render a video shot list
for "每日安全资讯 (2026-05-30)".

Features:
- Immutable Shot dataclass with full validation
- ShotList container with O(1) lookup and order preservation
- Export to Markdown and JSON
- Comprehensive logging and error handling
- Type-safe with strict annotations
- Security checks (URL validation, output sanitisation)
- Clean, performant, and maintainable code
"""

import json
import logging
import re
from collections import OrderedDict
from dataclasses import dataclass, field
from datetime import date
from pathlib import Path
from typing import Any, Dict, Iterator, List, Optional, Sequence, Tuple, Union

# ---------------------------------------------------------------------------
# Logging configuration (library‑style: no handler, let consumer configure)
# ---------------------------------------------------------------------------
logger = logging.getLogger(__name__)
logger.addHandler(logging.NullHandler())
logger.setLevel(logging.DEBUG)

# ---------------------------------------------------------------------------
# Custom exceptions
# ---------------------------------------------------------------------------
class ShotListError(Exception):
    """Base exception for shot list errors."""


class ValidationError(ShotListError):
    """Raised when a shot or metadata fails validation."""


class DuplicateShotError(ShotListError):
    """Raised when two shots share the same number."""


class ShotNotFoundError(ShotListError):
    """Raised when referencing a non‑existent shot."""


# ---------------------------------------------------------------------------
# Constants
# ---------------------------------------------------------------------------
REQUIRED_SHOT_KEYS = frozenset({
    "shot_number",
    "visual_description",
    "camera_movement",
    "graphics_overlay",
    "reference_image_urls",
})

VALID_CAMERA_MOVEMENTS = frozenset({
    "static", "zoom_in", "zoom_out", "pan_left", "pan_right",
    "tilt_up", "tilt_down", "custom",
})

MAX_SHOT_NUMBER = 999
DEFAULT_DATE = date(2026, 5, 30)

# URL validation regex (simple but production‑aware)
URL_PATTERN = re.compile(
    r"^https?://[^\s/$.?#].[^\s]*$",
    re.IGNORECASE,
)

# Markdown special characters that need escaping in plain text fields
MARKDOWN_ESCAPE_PATTERN = re.compile(r'([_*\[\]()#+\-!`~|<>\\])')

# ---------------------------------------------------------------------------
# Utility functions
# ---------------------------------------------------------------------------
def escape_markdown(text: str) -> str:
    """
    Escape Markdown special characters in a string to prevent injection.

    Args:
        text: Raw string potentially containing Markdown syntax.

    Returns:
        Escaped string safe for use in Markdown content.
    """
    return MARKDOWN_ESCAPE_PATTERN.sub(r'\\\1', text)


def validate_url(url: str) -> None:
    """
    Validate that a string is a well‑formed HTTP(S) URL.

    Args:
        url: URL string to validate.

    Raises:
        ValidationError: If the URL does not match the expected pattern.
    """
    if not isinstance(url, str):
        raise TypeError(f"URL must be a string, got {type(url).__name__}")
    if not URL_PATTERN.match(url):
        raise ValidationError(f"Invalid URL format: {url}")


def validate_camera_movement(movement: str) -> str:
    """
    Normalize and validate a camera movement descriptor.

    Args:
        movement: Raw camera movement string.

    Returns:
        Normalized movement string (lowercase with underscores).

    Raises:
        ValueError: If the movement is not in the allowed set.
    """
    cleaned = movement.lower().replace(" ", "_")
    if cleaned not in VALID_CAMERA_MOVEMENTS:
        valid = ", ".join(sorted(VALID_CAMERA_MOVEMENTS))
        raise ValueError(
            f"Invalid camera_movement '{movement}'. Must be one of: {valid}"
        )
    return cleaned


# ---------------------------------------------------------------------------
# Shot dataclass (immutable, validated)
# ---------------------------------------------------------------------------
@dataclass(frozen=True, order=True)
class Shot:
    """
    Represents a single shot in the video storyboard.

    Attributes:
        number: Unique shot number (1‑based, between 1 and 999).
        visual_description: Description of what appears on screen.
        camera_movement: Type of camera movement (must be in VALID_CAMERA_MOVEMENTS).
        graphics: Text overlays, animations, etc.
        references: Tuple of reference image URLs (optional, default empty).
    """

    number: int
    visual_description: str
    camera_movement: str
    graphics: str
    references: Tuple[str, ...] = field(default_factory=tuple, compare=False)

    def __post_init__(self) -> None:
        """Validate all fields after initialisation."""
        # Validate number
        if not isinstance(self.number, int):
            raise TypeError(f"number must be int, got {type(self.number).__name__}")
        if self.number < 1 or self.number > MAX_SHOT_NUMBER:
            raise ValueError(
                f"number must be between 1 and {MAX_SHOT_NUMBER}, got {self.number}"
            )

        # Validate visual_description
        if not isinstance(self.visual_description, str) or not self.visual_description.strip():
            raise ValueError("visual_description cannot be empty or whitespace")

        # Validate camera_movement (normalise to lower+underscore)
        cleaned = validate_camera_movement(self.camera_movement)
        # Since dataclass is frozen, we use object.__setattr__ to set the normalized value
        object.__setattr__(self, 'camera_movement', cleaned)

        # Validate graphics
        if not isinstance(self.graphics, str) or not self.graphics.strip():
            raise ValueError("graphics cannot be empty or whitespace")

        # Validate references (each must be a valid URL)
        if not isinstance(self.references, tuple):
            raise TypeError(f"references must be a tuple, got {type(self.references).__name__}")
        for url in self.references:
            validate_url(url)

        logger.debug("Shot %d created: %s", self.number, self.visual_description[:50])

    # -----------------------------------------------------------------------
    # Property: references as a mutable list (defensive copy)
    # -----------------------------------------------------------------------
    @property
    def reference_list(self) -> List[str]:
        """Return a mutable copy of the references tuple."""
        return list(self.references)

    # -----------------------------------------------------------------------
    # Factory method from dictionary
    # -----------------------------------------------------------------------
    @classmethod
    def from_dict(cls, data: Dict[str, Any]) -> "Shot":
        """
        Create a Shot from a dictionary (e.g., JSON input).

        Args:
            data: Dictionary containing shot keys.

        Returns:
            A new Shot instance.

        Raises:
            ValidationError: If required keys are missing or values invalid.
        """
        if not isinstance(data, dict):
            raise TypeError(f"data must be a dict, got {type(data).__name__}")

        missing = REQUIRED_SHOT_KEYS - set(data.keys())
        if missing:
            raise ValidationError(
                f"Missing required keys: {', '.join(sorted(missing))}"
            )

        try:
            # Map data keys to constructor parameters
            shot_data = {
                "number": data["shot_number"],
                "visual_description": data["visual_description"],
                "camera_movement": data["camera_movement"],
                "graphics": data["graphics_overlay"],
                "references": tuple(data.get("reference_image_urls", [])),
            }
            # Validate references list type
            if not isinstance(shot_data["references"], tuple):
                raise TypeError("reference_image_urls must be a list or tuple")
            return cls(**shot_data)
        except (ValueError, TypeError, ValidationError) as exc:
            logger.error("Failed to create Shot from dict: %s", exc)
            raise ValidationError(f"Invalid shot data: {exc}") from exc

    # -----------------------------------------------------------------------
    # Serialisation helpers
    # -----------------------------------------------------------------------
    def to_dict(self) -> Dict[str, Any]:
        """
        Convert shot to a plain dictionary (JSON serialisable).

        Returns:
            Dictionary representation of the shot.
        """
        return {
            "shot_number": self.number,
            "visual_description": self.visual_description,
            "camera_movement": self.camera_movement,
            "graphics_overlay": self.graphics,
            "reference_image_urls": list(self.references),
        }

    def to_markdown_row(self) -> str:
        """
        Generate a Markdown table row for this shot.

        Returns:
            String containing a single table row with pipes.
        """
        refs = ", ".join(self.references) if self.references else "—"
        # Escape fields for Markdown safety
        vis = escape_markdown(self.visual_description)
        cam = escape_markdown(self.camera_movement)
        gfx = escape_markdown(self.graphics)
        return f"| {self.number} | {vis} | {cam} | {gfx} | {refs} |"


# ---------------------------------------------------------------------------
# ShotList container — O(1) lookup, order preservation
# ---------------------------------------------------------------------------
class ShotList:
    """
    Ordered collection of Shots with fast number lookup.

    Internally uses an OrderedDict mapping shot number -> Shot.
    Maintains insertion order and provides list-like iteration.
    """

    def __init__(self, shots: Optional[Sequence[Shot]] = None) -> None:
        """
        Initialise a ShotList, optionally from an iterable of Shots.

        Args:
            shots: Initial sequence of Shot objects (may be empty).

        Raises:
            DuplicateShotError: If any shot number appears more than once.
        """
        self._shots: OrderedDict[int, Shot] = OrderedDict()
        if shots is not None:
            for shot in shots:
                self.add(shot)

    # -------------------------------------------------------------------
    # Core operations
    # -------------------------------------------------------------------
    def add(self, shot: Shot) -> None:
        """
        Add a shot to the list.

        Args:
            shot: Must be a valid Shot instance.

        Raises:
            TypeError: If shot is not a Shot.
            DuplicateShotError: If a shot with the same number already exists.
        """
        if not isinstance(shot, Shot):
            raise TypeError(f"Expected Shot instance, got {type(shot).__name__}")
        if shot.number in self._shots:
            raise DuplicateShotError(
                f"Shot number {shot.number} already exists"
            )
        self._shots[shot.number] = shot
        logger.debug("Added shot %d to ShotList", shot.number)

    def __getitem__(self, number: int) -> Shot:
        """
        O(1) lookup by shot number.

        Args:
            number: Shot number to retrieve.

        Returns:
            The corresponding Shot.

        Raises:
            ShotNotFoundError: If no shot with that number exists.
        """
        try:
            return self._shots[number]
        except KeyError:
            raise ShotNotFoundError(f"Shot number {number} not found") from None

    def __contains__(self, number: int) -> bool:
        """Check if a shot number exists in the list."""
        return number in self._shots

    def __len__(self) -> int:
        """Return the number of shots."""
        return len(self._shots)

    def __iter__(self) -> Iterator[Shot]:
        """Iterate over shots in insertion order."""
        return iter(self._shots.values())

    def __repr__(self) -> str:
        return f"ShotList({len(self)} shots)"

    # -------------------------------------------------------------------
    # Bulk operations
    # -------------------------------------------------------------------
    def get_all(self) -> List[Shot]:
        """Return all shots as a list (in order)."""
        return list(self._shots.values())

    def remove(self, number: int) -> None:
        """
        Remove a shot by number.

        Args:
            number: Shot number to remove.

        Raises:
            ShotNotFoundError: If shot does not exist.
        """
        if number not in self._shots:
            raise ShotNotFoundError(f"Cannot remove: shot {number} not found")
        del self._shots[number]
        logger.info("Removed shot %d", number)

    # -------------------------------------------------------------------
    # Serialisation
    # -------------------------------------------------------------------
    def to_markdown(self, title: str = "每日安全资讯 Shot List") -> str:
        """
        Export shot list as a Markdown document.

        Args:
            title: Title for the document (default: "每日安全资讯 Shot List").

        Returns:
            Complete Markdown string with table.
        """
        if not self._shots:
            return f"# {title}\n\n_No shots in list._\n"

        lines: List[str] = [
            f"# {title}",
            "",
            "| # | Visual Description | Camera Movement | Graphics Overlay | Reference Images |",
            "|---|---|---|---|---|",
        ]
        for shot in self:
            lines.append(shot.to_markdown_row())
        lines.append("")  # trailing newline
        return "\n".join(lines)

    def to_json(self, indent: int = 2) -> str:
        """
        Export shot list as a JSON string.

        Args:
            indent: Number of spaces for indentation (default: 2).

        Returns:
            JSON string representation.
        """
        data = [shot.to_dict() for shot in self]
        return json.dumps(data, indent=indent, ensure_ascii=False)

    def to_json_file(self, path: Path, indent: int = 2) -> None:
        """
        Write shot list to a JSON file.

        Args:
            path: Destination file path.
            indent: JSON indentation.

        Raises:
            OSError: If file cannot be written.
        """
        try:
            with open(path, "w", encoding="utf-8") as f:
                f.write(self.to_json(indent))
            logger.info("JSON written to %s", path)
        except OSError as exc:
            logger.error("Failed to write JSON file %s: %s", path, exc)
            raise

    # -------------------------------------------------------------------
    # Class methods for construction
    # -------------------------------------------------------------------
    @classmethod
    def from_dicts(cls, shot_dicts: List[Dict[str, Any]]) -> "ShotList":
        """
        Create a ShotList from a list of dictionaries.

        Args:
            shot_dicts: List of dictionaries, each convertible via Shot.from_dict.

        Returns:
            A new ShotList instance.

        Raises:
            ValidationError: If any dictionary is invalid.
        """
        shots = []
        for idx, d in enumerate(shot_dicts):
            try:
                shot = Shot.from_dict(d)
                shots.append(shot)
            except (ValidationError, TypeError) as exc:
                logger.error("Error in shot dict at index %d: %s", idx, exc)
                raise ValidationError(f"Invalid shot dict at index {idx}: {exc}") from exc
        return cls(shots)

    @classmethod
    def from_json(cls, json_str: str) -> "ShotList":
        """
        Deserialize a JSON string into a ShotList.

        Args:
            json_str: JSON string containing a list of shot dictionaries.

        Returns:
            A new ShotList.

        Raises:
            ValidationError: If JSON parsing fails or data is invalid.
        """
        try:
            data = json.loads(json_str)
        except json.JSONDecodeError as exc:
            raise ValidationError(f"Invalid JSON: {exc}") from exc

        if not isinstance(data, list):
            raise ValidationError("JSON must be a list of shot objects")

        return cls.from_dicts(data)

# Main entry point (for demonstration / CLI usage)
# ---------------------------------------------------------------------------
def main() -> None:
    """
    Demonstrate building and exporting the default daily shot list.
    Logs output paths and content previews.
    """

        level=logging.DEBUG,
        format="%(asctime)s [%(levelname)s] %(name)s: %(message)s",
    )
    logger.info("Starting daily security video shot list generator")

    # Build the shot list
    try:
        shot_list = build_default_shot_list()
    except ShotListError as exc:
        logger.critical("Aborting: %s", exc)
        return

    md_content = shot_list.to_markdown()
    print("=== Markdown Preview ===")
    print(md_content[:500] + "...\n")

    md_path = Path("daily_security_news_shots.md")
    try:
        md_path.write_text(md_content, encoding="utf-8")
        logger.info("Markdown written to %s", md_path.resolve())
    except OSError as exc:
        logger.error("Could not write Markdown file: %s", exc)

    # Export JSON
    json_content = shot_list.to_json()
    print("=== JSON Preview (first 600 chars) ===")
    print(json_content[:600] + "...\n")

    json_path = Path("daily_security_news_shots.json")
    shot_list.to_json_file(json_path)

    logger.info("All exports complete.")


if __name__ == "__main__":
